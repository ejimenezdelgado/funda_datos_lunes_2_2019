CREATE TABLE clientes 
  ( 
     codigo          SERIAL, 
     nombre          VARCHAR(30), 
     domicilio       VARCHAR(30), 
     ciudad          VARCHAR(20), 
     codigoprovincia SMALLINT NOT NULL, 
     PRIMARY KEY(codigo) 
  ); 

CREATE TABLE provincias 
  ( 
     codigo SERIAL, 
     nombre VARCHAR(20), 
     PRIMARY KEY (codigo) 
  );
