SELECT m.nombre, 
       m.edad, 
       v.nombre, 
       v.edad 
FROM   mujeres AS m 
       CROSS JOIN varones AS v; 

SELECT m.nombre, 
       m.edad, 
       v.nombre, 
       v.edad 
FROM   mujeres AS m 
       CROSS JOIN varones AS v 
WHERE  m.edad > 40 
       AND v.edad > 40; 

SELECT m.nombre, 
       m.edad, 
       v.nombre, 
       v.edad 
FROM   mujeres AS m 
       CROSS JOIN varones AS v
